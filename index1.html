Multi-Game Implementation Guide

Integration Steps

  1. SDK Installation

    npm install @exura/multi-game-sdk
yarn add @exura/multi-game-sdk

  2. Initialize Multi-Game SDK

    import { ExuraMultiGame } from '@exura/multi-game-sdk';

const exura = new ExuraMultiGame({
    network: 'mainnet-beta',
    supportedGames: ['tibia', 'runescape', 'wow'],
    apiKey: 'YOUR_API_KEY'
});

  3. Game-Specific Setup

    // Configure game connections
await exura.configureGame({
    game: 'tibia',
    apiEndpoint: 'https://api.tibia.com',
    credentials: {
        // game-specific credentials
    }
});

Multi-Game API Reference

// Initialize Exura client
const exura = new ExuraClient({
    network: 'mainnet-beta',
    wallet: yourWallet
});

// Multi-game character verification
const character = await exura.verifyGameCharacter({
    game: 'tibia', // or 'runescape', 'wow'
    name: 'PlayerName',
    server: 'ServerName'
});

// Cross-game swap
await exura.crossGameSwap({
    fromGame: 'tibia',
    toGame: 'runescape',
    fromAsset: 'TIBIA_COINS',
    toAsset: 'OSRS_GOLD',
    amount: 100,
    character: character.name
});

Universal Security Features

// Multi-game security implementation
const securityConfig = {
    tibia: {
        maxTransaction: '1000000 TC',
        verificationLevel: 'HIGH',
        timelock: '24h'
    },
    runescape: {
        maxTransaction: '100M OSRS',
        verificationLevel: 'MEDIUM',
        timelock: '12h'
    },
    wow: {
        maxTransaction: '10M GOLD',
        verificationLevel: 'HIGH',
        timelock: '6h'
    }
};

// Verify cross-game transaction
async function verifyCrossGameTx(fromGame, toGame, amount) {
    const security = await exura.security.verifyCrossGame({
        sourceGame: fromGame,
        targetGame: toGame,
        amount: amount,
        config: securityConfig[fromGame]
    });
    
    return security.verified;
}
